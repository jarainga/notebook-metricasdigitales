import matplotlib.pyplot as plt

# Definir colores específicos para ciertos entrevistados
colores_especificos = {
    "Paolo Guerrero": "green",
    "Andre Carrillo": "skyblue",
    "Juan Manuel Loco Vargas": "goldenrod",
    "Juan Manuel Loco Vargas 2Da": "goldenrod"
}

# Lista de extranjeros que deben estar en rojo
extrajeros = ["Juanfer Quintero", "Faustino Asprilla", "Edwin Cardona", "Rafinha", "Faryd Mondragon"]

# Crear la figura del gráfico
plt.figure(figsize=(14, 7))

# Graficar todas las líneas en azul por defecto
plt.plot(base2["Fecha_Estandar"], base2["Vistas"], marker="o", linestyle="-", color="b", linewidth=2)

# Etiquetar todos los puntos con sus entrevistados y aplicar colores específicos
for fecha, vistas, entrevistado in zip(base2["Fecha_Estandar"], base2["Vistas"], base2["Entrevistado"]):
    # Asignar color según las condiciones
    if entrevistado in colores_especificos:
        color = colores_especificos[entrevistado]
    elif entrevistado in extrajeros:
        color = "red"
    else:
        color = "black"

    fontweight = "bold" if entrevistado in extrajeros or entrevistado in colores_especificos else "normal"
    
    plt.annotate(entrevistado, (fecha, vistas), textcoords="offset points", xytext=(0,10), 
                 ha='center', fontsize=8, rotation=90, color=color, fontweight=fontweight)

# Personalización del gráfico
plt.title("Canal de Youtube JF10 TV - Evolución de Vistas 2024-2025", fontsize=16, fontweight="bold")
plt.xlabel("Fecha", fontsize=14)
plt.ylabel("Vistas", fontsize=14)
plt.xticks(rotation=90, fontsize=12)
plt.yticks(fontsize=12)
plt.grid(True, linestyle="--", alpha=0.5)

# Mostrar el gráfico
plt.show()
